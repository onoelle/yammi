
# http://cvs2svn.tigris.org/cvs2git.html

mkdir yammi-cvs-rsync
cd yammi-cvs-rsync
rsync -av rsync://yammi.cvs.sourceforge.net/cvsroot/yammi/ .
cvs2git --options=cvs2git-yammi.options --fallback-encoding=ISO-8859-15
cd ..

mkdir yammi.git
cd yammi.git
git init

cat ../yammi-cvs-rsync/cvs2svn-tmp/git-blob.dat ../yammi-cvs-rsync/cvs2svn-tmp/git-dump.dat | git fast-import

git gc --prune=now
git checkout
